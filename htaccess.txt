<IfModule mod_rewrite.c>

	# Start Rewrite Engine
	RewriteEngine on
	
	# DNS LOOKUP API RULES
	
	# Match for Domain and Update Request
	RewriteRule ^lookup/domain/(.*)/update$ lookup.php?domain=$1&update=update [QSA,L]
	
	# Match for Domain Only
	RewriteRule ^lookup/domain/(.*)$ lookup.php?domain=$1 [QSA,L]
	
	# Match for IP
	RewriteRule ^lookup/ip/(.*)$ lookup.php?ip=$1 [QSA,L]
	
	# Match without Domain
	RewriteRule ^lookup/domain/?$ lookup.php [QSA,L]
	
	# Match without IP
	RewriteRule ^lookup/ip/?$ lookup.php [QSA,L]
	
	# Match for A record lookup with hostname
	RewriteRule ^lookup/a/(.*)$ lookup.php?a=$1 [QSA,L]
	
	# Match for A record lookup without hostname
	RewriteRule ^lookup/a/?$ lookup.php [QSA,L]
	
	# Match for AAAA record lookup with hostname
	RewriteRule ^lookup/aaaa/(.*)$ lookup.php?aaaa=$1 [QSA,L]
	
	# Match for AAAA record lookup without hostname
	RewriteRule ^lookup/aaaa/?$ lookup.php [QSA,L]
	
	# Match for MX record lookup with hostname
	RewriteRule ^lookup/mx/(.*)$ lookup.php?mx=$1 [QSA,L]
	
	# Match for MX record lookup without hostname
	RewriteRule ^lookup/mx/?$ lookup.php [QSA,L]
	
	# Match for TXT record lookup with hostname
	RewriteRule ^lookup/txt/(.*)$ lookup.php?txt=$1 [QSA,L]
	
	# Match for TXT record lookup without hostname
	RewriteRule ^lookup/txt/?$ lookup.php [QSA,L]
	
	# Backwards Compatibility Match for Domain and Update Request
	RewriteRule ^lookup/(.*)/update$ lookup.php?domain=$1&update=update [QSA,L]
	
	# Backwards Compatibility Match for Domain Only
	RewriteRule ^lookup/(.*)$ lookup.php?domain=$1 [QSA,L]
	
	# Match without Domain or IP
	RewriteRule ^lookup/?$ lookup.php [QSA,L]
	
	# END DNS LOOKUP API RULES
	
</IfModule>
